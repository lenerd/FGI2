\begin{enumerate}
    \item
        Die Menge aller $T$-Invarianten-Vektoren ist die Menge $J$ der Lösungen
        $j$ des folgenden Gleichungssystems.
        \begin{equation}
            \begin{split}
                \Delta_{N_{10.3}} \cdot \textbf{j} &= 0 \\
                \begin{pmatrix}
                    -1 & -1 &  3 &  2 & -1 \\
                     0 & -1 &  3 &  0 &  0 \\
                     0 &  1 & -3 &  0 &  0 \\
                     1 &  2 & -6 & -2 &  1 \\
                \end{pmatrix}
                \cdot
                \begin{pmatrix}
                    j_1 \\ j_2 \\ j_3 \\ j_4 \\ j_5
                \end{pmatrix}
                &=
                \begin{pmatrix}
                    0 \\ 0 \\ 0 \\ 0
                \end{pmatrix}
            \end{split}
        \end{equation}
        \begin{equation}
            J = \left\{
                \textbf{j} = \left( j_1, j_2, j_3, j_4, j_5 \right)^T
                \in \mathbb{N}^5
                \ |\  j_1 - 2j_2 + j_5 = 0 \land j_2 - j_3 = 0
                \right\}
        \end{equation}
        
    \item
        \begin{align}
            \textbf{j} &= \left( 1, 3, 1, 1, 1 \right)^T \\
